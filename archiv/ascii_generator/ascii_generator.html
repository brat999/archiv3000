DIY ASCII generator zine

Der Generator wurde mit Python programmiert und basiert auf ASCII (American Standard Code for Information Interchange). Die 7-Bit-Zeichenkodierung enthält unter anderem 95 druckbare Zeichen.

Das Zine entstand in Verbindung mit dem Projekt "Time Out for Some Zine Love" an der F+F Schule für Kunst und Design, mit der Aufgabenstellung eine persönliche Sammlung zu publizieren.

Den Code für den Generator findest du am Ende dieser Seite. Für die optimale Darstellung in einem Textdokument müssen die Zeichen mit einer mono spaced font gesetzt sein.

copy paste the python code here:

import PIL.Image

# ASCII characters used to build the output text

ASCII_CHARS = ["‹","!","#","Ç","-","_","#","?","“","≠"," "]

# resize image according to a new width

def resize_image(image, new_width=100):

width, height = image.size

ratio = height / width

new_height = int(new_width * ratio)

resized_image = image.resize((new_width, new_height))

return(resized_image)

# convert each pixel to grayscale

def grayify(image):

grayscale_image = image.convert("L")

return(grayscale_image)

# convert pixels to a string of ASCII characters

def pixels_to_ascii(image):

pixels = image.getdata()

characters = "".join([ASCII_CHARS[pixel//25] for pixel in pixels])

return(characters)

def main(new_width=100):

# attempt to open image from user-input

path = input("enter a valid pathname to an image:\n")

try:

image = PIL.Image.open(path)

except:

print(path, "is not a valid pathname to an image.")

return

# convert image to ASCII

new_image_data = pixels_to_ascii(grayify(resize_image(image)))

# format

pixel_count = len(new_image_data)

ascii_image = "\n".join(new_image_data[i:(i+new_width)] for i in range(0, pixel_count, new_width))

# print result

print(ascii_image)

# save result to "ascii_image.txt"

with open("ascii_image.txt", "w") as f:

f.write(ascii_image)

# Start the program

if __name__ == "__main__":

main()
